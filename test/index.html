class Openshift:

    __logging_separator = "=" * 100

    def __init__(self, openshift_url, username, password):
        self.__logger = log.Logging("Openshift module")
        self._openshift_url = openshift_url
        self._username = username
        self._password = password
        self._token = self.gettingToken()
        self._headers = {"Authorization": 'Bearer ' + self._token
                         + " Accept: application/json"}
        self._headers2 = {"Authorization": 'Bearer ' + self._token,
                          "Accept": "application/json",
                          "Content-Type": "application/strategic-merge-patch+json"}

    # Funcion para obtener el Token de autenticacion en Openshift

    def gettingToken(self):

        self.__logger.info("Getting Token " + self._openshift_url)
        url = self._openshift_url + '/oauth/authorize?response_type=token&client_id=openshift-challenging-client'
        r = requests.get(url, auth=(self._username, self._password),
                         verify=False, allow_redirects=True)
        if(r.status_code != 302 and r.status_code != 200):
            self.__logger.error("Error getting token")
            raise OpenShiftLoginException(r.status_code)
        r1 = re.search('^.*access_token=(.*?)&.*', r.url)
        token = r1.groups()[0]
        return token


    def getSecret(self, SecretName, namespace):

        """
        Funcion para crear secret, parametros a introducir:

            - secretName: nombre del secret a crear
            - namespace: Proyecto en el que crear el secret,incluimos entorno
            - datajson: json de creacion del secret
                Ejemplo:
                        {
                              "apiVersion": "v1",
                              "data": {
                                "db2.datasource.password": "xxxxxxxxxx",
                                "db2.datasource.username": "xxxxxxxxxx",
                              },
                              "kind": "Secret",
                              "metadata": {
                                            "name": "application",
                                            "namespace": self._namespace
                              },
                              "type": "Opaque"
                            }
        """

        self.__logger.info("Getting Secret " + SecretName
                           + " info in " + namespace)
        url = self._openshift_url + "/api/v1/namespaces/" + namespace \
            + "/secrets/" + SecretName
        r = requests.get(url, headers=self._headers, verify=False)
        return r.text

    def createSecret(self, secretName, namespace, datajson={}):

        self.__logger.info("Creating Secret in " + namespace)
        url = self._openshift_url + "/api/v1/namespaces/" + namespace \
            + "/secrets"
        r = requests.post(url, data=json.dumps(datajson),
                          headers=self._headers, verify=False)
        return r.status_code

def gettingToken(self):
    self.__logger.info("Getting Token " + self._openshift_url)
    url = self._openshift_url + '/oauth/authorize?response_type=token&client_id=openshift-challenging-client'
    r = requests.get(url, auth=(self._username, self._password),
                     verify=False, allow_redirects=True)
    if(r.status_code != 302 and r.status_code != 200):
        self.__logger.error("Error getting token")
        raise OpenShiftLoginException(r.status_code)
    r1 = re.search('^.*access_token=(.*?)&.*', r.url)
    token = r1.groups()[0]
    return token
